let compare = fun (x:t) (y:t) ->
 match x with
 | Saturation ->
  begin
   match y with
   | Saturation -> 0
   | _ -> -1
  end
 | Witness ->
  begin
   match y with
   | Witness -> 0
   | _ -> 1
  end
 | Index i ->
  begin
   match y with
   | Saturation -> -1
   | Witness -> 1
   | Index j -> Pervasives.compare i j 
   | Element yy -> Pervasives.compare i ( to_int y )
  end
 | Element xx ->
  begin
   match y with
   | Saturation -> -1
   | Witness -> 1
   | Index j -> Pervasives.compare ( to_int x ) j
   | Element yy -> Pervasives.compare ( to_int x ) ( to_int y )
  end
