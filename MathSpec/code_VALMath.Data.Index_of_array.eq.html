let eq = fun (x:t) (y:t) ->
 match x with
 | Saturation ->
  begin
   match y with
   | Saturation -> true
   | _ -> false
  end
 | Witness ->
  begin
   match y with
   | Witness -> true
   | _ -> false
  end
 | Index i ->
  begin
   match y with
   | Saturation | Witness -> false
   | Index j -> i = j
   | Element yy -> A.eq ( A.a.(i) ) yy
  end
 | Element xx ->
  begin
   match y with
   | Saturation | Witness -> false
   | Index j -> A.eq ( A.a.(j) ) xx
   | Element yy -> A.eq xx yy
  end
