let float_int_decr_rect = fun (ratio:float) (n:int) (f:float -> float) (a:float) (b:float) ->
 let accu = ref 0.
 and x = ref a
 and length = ref ( ( b -. a ) *. ( 1. -. ratio ) /. ( 1. -. ratio ** ( float n ) ) ) in
  for i = 0 to n - 1 do
   x := !x +. !length ;
   accu := !accu +. ( f !x ) *. !length ;
   length := ratio *. !length ;
  done ;
  !accu
