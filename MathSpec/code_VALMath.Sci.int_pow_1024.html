let rec int_pow_1024 = fun (n:int) (a:Num.num array) -> match n with
 | 0 -> sci_1
 | 1 -> a
 | -1 -> inv_1024 a
 | _ ->
  begin
   if n < 0 then int_pow_1024 ( abs n ) ( inv_1024 a )
   else
    begin
     let b = int_pow_1024 ( n / 2 ) a in
      if n mod 2 = 0 then mult_1024 b b
      else mult_1024 ( mult_1024 b b ) a
    end
  end
