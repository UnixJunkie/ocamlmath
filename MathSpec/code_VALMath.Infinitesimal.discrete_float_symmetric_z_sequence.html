let discrete_float_symmetric_z_sequence = fun (v:float array) (z:float array) ->
 let n = Array.length v
 and coeff = 1. /. ( z.(0) *. z.(0) +. z.(1) *. z.(1) )
 and z0 = ref 1.
 and z1 = ref 0.
 and zz0 = ref 1.
 and zz1 = ref 0. in
  let w = Array.make_matrix n 2 0.
  and nn = n / 2
  and y = [| coeff *. z.(0) ; -. coeff *. z.(1) |] in
    for i = nn to ( n - 1 ) do
     let row = w.(i)
     and vv = v.(i) in
      row.(0) <- vv *. !z0 ;
      row.(1) <- vv *. !z1 ;
      w.(i) <- row ;
      zz0 := !z0 *. y.(0) -. !z1 *. y.(1) ;
      zz1 := ( !z1 *. y.(0) ) +. ( !z0 *. y.(1) ) ;
      z0 := !zz0 ;
      z1 := !zz1 ;
    done ;
    z0 := z.(0) ;
    z1 := z.(1) ;
    zz0 := z.(0) ;
    zz1 := z.(1) ;
    for i = ( nn - 1 ) downto 0 do
     let row = w.(i)
     and vv = v.(i) in
      row.(0) <- vv *. !z0 ;
      row.(1) <- vv *. !z1 ;
      w.(i) <- row ;
      zz0 := !z0 *. z.(0) -. !z1 *. z.(1) ;
      zz1 := ( !z1 *. z.(0) ) +. ( !z0 *. z.(1) ) ;
      z0 := !zz0 ;
      z1 := !zz1 ;
    done ;
    w
