let read_bmp_header = function (fname:string) ->
 let ic = open_in fname
 and a = ref [] in
  begin
   try
    begin
     for i = 0 to 53 do
      a := ( input_byte ic ) :: !a
     done ;
     close_in_noerr ic ;
    end
   with _ ->
    close_in_noerr ic
  end ;
  let b = Array.of_list ( List.rev !a ) in
   if ( char_of_int b.(0) <> 'B' ) || ( char_of_int b.(1) <> 'M' ) then
    failwith "Bad magic number in Readwrite.read_bmp_header." ;
   let c = Array.make 16 "BM" in
    c.(1) <- f32 2 b ;
    c.(2) <- f16 6 b ;
    c.(3) <- f16 8 b ;
    c.(4) <- f32 10 b ;
    c.(5) <- f32 14 b ;
    c.(6) <- f32 18 b ;
    c.(7) <- f32 22 b ;
    c.(8) <- f16 26 b ;
    c.(9) <- f16 28 b ;

    c.(12) <- f32 38 b ;
    c.(13) <- f32 42 b ;
