let float_weighted_int_minus1_1 = fun (x:float array) (w:float array) (f:float -> float) (a:float) (b:float) ->
 let l = Array.length x
 and accu = ref 0.
 and center = ( a +. b ) /. 2.
 and halflength = ( b -. a ) /. 2. in
  for i = 0 to l - 1 do
   accu := !accu +. w.(i) *. f ( center +. halflength *. x.(i) ) ;
  done ;
  !accu *. halflength
