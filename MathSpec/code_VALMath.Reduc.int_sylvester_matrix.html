let int_sylvester_matrix = fun (p:int array) (q:int array) ->
 let dp = int_of_float ( poly_int_deg p )
 and dq = int_of_float ( poly_int_deg q ) in
  let dd = dp + dq in
   let m = Array.make_matrix dd dd 0 in
    for i = 0 to pred dq do
     let row = m.(i)
     and ii = i + dp in
      for j = i to ii do
       row.(j) <- p.( ii - j ) ;
      done ;
    done ;
    for i = 0 to pred dp do
     let ii = i + dq in
      let row = m.(ii) in
       for j = i to ii do
        row.(j) <- q.( ii - j ) ;
       done ;
    done ;
    m
