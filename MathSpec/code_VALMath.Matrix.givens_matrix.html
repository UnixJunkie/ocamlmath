let givens_matrix = fun (r:int) (i:int) (j:int) (t:float) ->
 let w = identity_float r r in
  if i < j && j < r then
   begin
    let row = w.(i)
    and ligne = w.(j)
    and c = cos t
    and s = sin t in 
     row.(i) <- c ;
     row.(j) <- -. s ;
     ligne.(i) <- s ;
     ligne.(j) <- c ;
   end ;
  if j < i && i < r then
   begin
    let row = w.(i)
    and ligne = w.(j)
    and c = cos t
    and s = sin t in 
     row.(i) <- c ;
     row.(j) <- s ;
     ligne.(i) <- -. s ;
     ligne.(j) <- c ;
   end ;
   w
