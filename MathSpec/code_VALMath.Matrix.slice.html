let slice = fun (n:int) v ->
 let l = Array.length v in
  if l mod n != 0 then failwith "The number of lines must divide the length of the vector in Matrix.slice." ;
   let c = l / n in
    let m = Array.make_matrix n c v.(0)
    and cc =pred c in
     for i = 0 to pred n do
      let row = m.(i)
      and ni = n * i in
       for j = 0 to cc do
        row.(j) <- v.( ni + j )
       done
     done ;
     m
