 match x with
 | (Foa_matrix_cons y) ->
  begin
   let r = numrows y
   and cc = (numcolumns y) - 1 in
    let m_m = Array.make_matrix r (cc + 1) (Float_cons 0.) in
     for i = 0 to r - 1 do
      for j = 0 to cc do
       m_m.(i).(j) <- matrix_foa_copy y.(i).(j)
      done
     done ;
     m_m
  end
 | _ -> failwith "Bad type Foa_matrix_cons in Matrix.matrix_foa_demakeup."
