let matrix_complex_extract_column_to_matrix_trans = fun (j:int) m ->
 let rr = Array.length m in
  let r = rr / 2
  and x = Array.make rr m.(0).(0)
  and y = Array.make rr m.(0).(0) in
   let jj = ( Array.length m.(0) / 2 ) + j in
    for i = 0 to r - 1 do
     let first_row = m.(i)
     and ii = r + i in
      let second_row = m.(ii) in
       x.(i) <- first_row.(j) ;
       x.(ii) <- second_row.(j) ;
       y.(i) <- first_row.(jj) ;
       y.(ii) <- second_row.(jj) ;
    done ;
    [| x ; y |]
