let matrix_ioa_permu = fun (n:int) (i:int) (j:int) ->
 let w = Array.make_matrix n n (Int_cons 0)
 and ii = min i j
 and jj = Util.int_max i j in
  for k = 0 to ii - 1 do
   w.(k).(k) <- Int_cons 1
  done ;
  w.(ii).(jj) <- Int_cons 1 ;
  for k = ii + 1 to jj - 1 do
   w.(k).(k) <- Int_cons 1
  done ;
  w.(jj).(ii) <- Int_cons 1 ;
  for k = jj + 1 to n do
   w.(k).(k) <- Int_cons 1
  done ;
  Ioa_matrix_cons w
