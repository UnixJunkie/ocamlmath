let float_laplace_complex = fun methode (b:float) (f:float -> float) (p:float array) ->
 let r = -. p.(0)
 and omega = -. p.(1) in
  let g = function x -> ( f x ) *. exp ( r *. x ) *. cos ( omega *. x )
  and h = function x -> ( f x ) *. exp ( r *. x ) *. sin ( omega *. x ) in
   [| methode g 0. b ; methode h 0. b |]
