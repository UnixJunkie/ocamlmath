let what = function (x:string) ->
 try
  what_math "../math" x
 with _ ->
  begin
   try
    what_math ( ( Filename.dirname Sys.executable_name ) ^ "/.." ) x
   with _ ->
    begin
     try
      what_math ( ( Filename.dirname Sys.executable_name ) ^ "/../../../doc/ocamlmath" ) x
     with _ ->
      begin
       try
        what_standard x
       with _ ->
        begin
         try
          what_findlib x
         with _ ->
          begin
           try
            what_ocamlgraph x
           with _ -> failwith "Not found in Spec.what."
          end
        end
      end
    end
  end
