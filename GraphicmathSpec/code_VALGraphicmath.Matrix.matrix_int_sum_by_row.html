let matrix_int_sum_by_row = function (m:int array array) ->
 let accu = Array.make (Array.length m) 0 in
  for i = 0 to (Array.length m) - 1 do
   let row = m.(i)
   and accumul = ref accu.(i) in
    for j = 0 to (Array.length row) - 1 do
     accumul := row.(j) + !accumul
    done ;
    accu.(i) <- !accumul
  done ;
  accu
