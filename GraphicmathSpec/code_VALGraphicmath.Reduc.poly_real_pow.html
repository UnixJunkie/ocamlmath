let rec poly_real_pow = fun mult_rule (n:int) (p:float array) ->
 match n with
  | 0 -> [| 1. |]
  | 1 -> p
  | 2 -> mult_rule p p
  | _ ->
   begin
    let nn = n / 2 in
     let pp = poly_real_pow mult_rule nn p in
      let prod = mult_rule pp pp in
       if n mod 2 = 0 then
        prod
       else
        mult_rule prod p
   end
