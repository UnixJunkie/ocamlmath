let triangle_barycentre = fun (weight:float) (x:int * int) (y:int * int) (z:int * int) ->
 [| int_of_float ( weight *. ( float ( fst x ) ) +. ( 0.5 *. ( 1. -. weight ) ) *. ( float ( fst y + fst z ) ) ) ;
  int_of_float ( weight *. ( float ( snd x ) ) +. ( 0.5 *. ( 1. -. weight ) ) *. ( float ( snd y + snd z ) ) ) |]
