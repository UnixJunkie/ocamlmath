let matrix_int_min_by_column = function (m:int array array) ->
 let cc = (numcolumns m) - 1 in
  let accu = Array.make (cc + 1) (- min_int) in
   for j = 0 to cc do
    let accumul = ref accu.(j) in
     for i = 0 to ( (Array.length m) - 1 ) do
      accumul := min m.(i).(j) !accumul
     done ;
     accu.(j) <- !accumul
   done ;
  accu
