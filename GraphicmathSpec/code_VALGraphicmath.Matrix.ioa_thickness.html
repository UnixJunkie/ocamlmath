let rec ioa_thickness = function (x:int_or_array) ->
 match x with
 | (Int_cons y) -> 0
 | (Int_vector_cons y) -> 0
 | (Int_matrix_cons y) -> 0 
 | (Ioa_vector_cons y) ->
  begin
   let accu = ref 0 in
    for i = 0 to (Array.length y) - 1 do
     let x = ioa_thickness y.(i) in
      if x > !accu then accu := x
    done ;
    1 + !accu
  end
 | (Ioa_matrix_cons y) ->
  begin
   let accu = ref 0 in
    for i = 0 to (Array.length y) - 1 do
     let row = y.(i) in
      for j = 0 to (Array.length row) - 1 do
       let x = ioa_thickness row.(j) in
        if x > !accu then accu := x
      done
    done ;
    1 + !accu
  end
 | _ -> 0
