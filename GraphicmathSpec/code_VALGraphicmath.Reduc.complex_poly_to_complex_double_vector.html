let complex_poly_to_complex_double_vector = function (p:float array array array) ->
 let l = Array.length p in
  let v = Array.make ( 2 * l ) 0.
  and vv = Array.make ( 2 * l ) 0. in
   for i = 0 to l - 1 do
    v.(i) <- p.(i).(0).(0) ;
    v.( l + i ) <- p.(i).(1).(0) ;
    vv.(i) <- p.(i).(0).(1) ;
    vv.( l + i ) <- p.(i).(1).(1) ;
   done ;
   [| v ; vv |]
