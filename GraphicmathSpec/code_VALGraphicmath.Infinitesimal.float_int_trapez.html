let float_int_trapez = fun (n:int) (f:float -> float) (a:float) (b:float) ->
 let accu = ref 0.
 and length = ( b -. a ) /. ( float n )
 and nn = n - 1 in
  for i = 1 to nn do
   accu := !accu +. ( f ( a +. ( float i ) *. length ) ) *. length
  done ;
  accu := !accu +. ( f a +. f b ) *. length *. 0.5 ;
  !accu
