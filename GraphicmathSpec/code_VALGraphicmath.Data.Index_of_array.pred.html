let rec pred = function (x:t) ->
 match x with
 | Saturation -> Saturation
 | Witness -> Witness
 | Index i ->
  begin
   match ( i < length , i > 0 ) with
   | ( true , true ) -> Index ( Pervasives.pred i )
   | ( false , _ ) -> Saturation
   | ( true , false ) -> Witness
  end
 | Element y -> pred ( Index ( to_int x ) )
