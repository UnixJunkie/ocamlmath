let float_end_ode_euler = fun (f:float -> float -> float) (nsteps:int) (y0:float) (beginning:float) (ending:float) ->
 let step = ( ending -. beginning ) /. ( float nsteps )
 and x = ref beginning
 and y = ref y0 in
  for i = 1 to nsteps do
   y := !y +. step *. f !x !y ;
   x := !x +. step ;
  done ;
  !y
