 let h_size = string_of_int size.(0)

 and v_real = coordinates.(3) -. coordinates.(2) in
  let h_margin = max ( size.(0) / 10 ) ( 10 * tsh )
  and v_margin = max ( size.(1) / 10 ) ( 4 * tsv ) in

     Graphics.open_graph ( " " ^ h_size ^ "x" ^ v_size ) ;
