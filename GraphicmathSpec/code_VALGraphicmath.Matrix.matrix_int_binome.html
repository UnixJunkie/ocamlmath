let rec matrix_int_binome = function n ->
 match n with
 | 0 -> [|[|1|]|]
 | 1 -> [| [|1|] ; [| 1 ; 1 |] |]
 | _ -> let m = matrix_int_binome (n - 1)
  and v = Array.make ( n + 1 ) 1 in
   let row = m.( n - 1 ) in
    for i = 1 to n - 1 do
     v.(i) <- row.(i - 1) + row.(i)
    done ;
    Array.append m [| v |]
