 let red = m.(0)
 and green = m.(1)
 and blue = m.(2) in
  let l = Array.length blue
  and c = Array.length blue.(0) in
   let w = Array.make_matrix l c 0
   and cc = pred c in
    for i = 0 to pred l do
     let row_output = w.(i)
     and row_red = red.(i)
     and row_green = green.(i)
     and row_blue = blue.(i) in
      for j = 0 to cc do
       row_output.(j) <- row_blue.(j) + 256 * ( row_green.(j) + 256 * row_red.(j) )
      done
    done ;
