let float_dirty_inv = function (m:float array array) ->
 let r = Array.length m in
  let mm = Array.make_matrix r r 0.
  and rr = pred r in
   for i = 0 to rr do
    let row_input = m.(i)
    and row_output = mm.(i) in
     for j = 0 to i - 1 do
      row_output.(j) <- -. row_input.(j)
     done ;
     row_output.(i) <- 2. -. row_input.(i) ;
     for j = i + 1 to rr do
      row_output.(j) <- -. row_input.(j)
     done ;
     mm.(i) <- row_output
   done ;
   mm
