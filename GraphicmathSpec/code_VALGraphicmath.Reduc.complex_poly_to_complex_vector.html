let complex_poly_to_complex_vector = function (p:float array array array) ->
 let l = Array.length p in
  let v = Array.make ( 2 * l ) 0. in
   for i = 0 to l - 1 do
    v.(i) <- 0.5 *. ( p.(i).(0).(0) +. p.(i).(1).(1) ) ;
    v.( l + i ) <- 0.5 *. ( p.(i).(1).(0) -. p.(i).(0).(1) ) ;
   done ;
   v
