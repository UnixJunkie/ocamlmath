let matrix_float_serret_frenet = function (v:float array) ->
 let r = Array.length v in
  let rr = r + 1
  and r_r = r - 1 in
   let m = Array.make_matrix rr rr 0. in
    m.(0).(1) <- v.(0) ;
    for i = 1 to r_r do
     let row = m.(i) in
      row.( i - 1 ) <- -. v.( i - 1 ) ;
      row.( i + 1 ) <- v.(i) ;
    done ;
    m.(r).(r_r) <- -. v.(r_r) ;
    m
